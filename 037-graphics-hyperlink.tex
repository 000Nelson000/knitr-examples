\begin{document}



Create link texts in the data:

\begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hlfunctioncall{library}(lattice)
\hlfunctioncall{data}(iris)
iris$hyperlink <- \hlfunctioncall{paste}(\hlstring{"\textbackslash{}\textbackslash{}hyperlink\{"}, iris$Species, \hlstring{"\}\{"}, iris$Species, 
    \hlstring{"\}"}, sep = \hlstring{""})
iris$hypertarget <- \hlfunctioncall{paste}(\hlstring{"\textbackslash{}\textbackslash{}hypertarget\{"}, iris$Species, \hlstring{"\}\{"}, iris$Species, 
    \hlstring{"\}"}, sep = \hlstring{""})
\end{alltt}
\end{kframe}
\end{knitrout}


Draw these texts in plots (you can click on the labels):

\begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\begin{kframe}
\begin{alltt}
\hlfunctioncall{xyplot}(Sepal.Length ~ Sepal.Width | hyperlink, data = iris)
\end{alltt}
\end{kframe}\input{figure/037-graphics-hyperlink-pltFacet.tikz}
\end{knitrout}


\begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\begin{kframe}
\begin{alltt}
\hlfunctioncall{xyplot}(Sepal.Length ~ Sepal.Width | hypertarget, data = iris, layout = \hlfunctioncall{c}(1, 
    1))
\end{alltt}
\end{kframe}\input{figure/037-graphics-hyperlink-pltInd1.tikz}\input{figure/037-graphics-hyperlink-pltInd2.tikz}\input{figure/037-graphics-hyperlink-pltInd3.tikz}
\end{knitrout}


Note the option external has to be FALSE here, otherwise LaTeX will not be able 
to find the link target in the document.
